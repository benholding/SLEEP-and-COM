##setting only population mean and SD, others are default non-informative

#Running the bayesian 't-test'
FAS_best <- BESTmcmc(VF.complete$Score.FAS[VF.complete$SD ==1], 
                     VF.complete$Score.FAS[VF.complete$SD ==0],
